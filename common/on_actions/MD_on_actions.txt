# This = construction ship
# Fires when construction is complete,
# immediately before station is created
on_building_starbase_exd_0 = {
	events = {
		magical_discoveries_crisis.1300
		magical_discoveries_crisis.1310
	}
}

on_monthly_pulse = {
	events = {
		md_market.1
		md_market.2
		md_market.3
		md_market.4
		magical_discoveries.101
	}
	random_events = {
		10 = magical_discoveries_crisis.0
		40 = 0
	}
}

# No scope, like on_game_start
on_yearly_pulse = {
	events = {
		magical_discoveries_crisis.1254
	}
}

# This = owner of ship 1 (destroyed)
# From = owner of ship 2 (combatant)
# FromFrom = ship 1
# FromFromFrom = ship 2

		magical_discoveries_crisis.1282
		magical_discoveries_crisis.1012
		magical_discoveries_crisis.1311
	}
}
