	picture = GFX_evt_ship_in_orbit
	hide_window = no
	## イベントの発生条件
	is_triggered_only = no
	fire_only_once = no
	mean_time_to_happen = {
		years = 15
	}
	trigger = {
		NOT = {
			has_country_flag = MeetMagicFlag
		}
		NOT = {
			has_country_flag = DenyMagicFlag
		}
	}
	## イベント発生時の効果
	immediate = {
	}
	## イベントの効果
	option = {
		name = magical_discoveries.1a.name
		set_country_flag = MeetMagicFlag
	}
	option = {
		name = magical_discoveries.1b.name
		set_country_flag = MeetMagicFlag
		trigger = {
			OR = {
				has_ethic = ethic_materialist
				has_ethic = ethic_fanatic_materialist
			}
		}
	}

		name = magical_discoveries.1c.name

country_event = {
	## イベントの基本情報
	id = magical_discoveries.8
	title = magical_discoveries.8.name
	desc = magical_discoveries.8.desc
	picture = GFX_evt_ship_in_orbit
	hide_window = no
	## イベントの発生条件
	is_triggered_only = no
	fire_only_once = no
	mean_time_to_happen = {
		years = 15
	}
	trigger = {
		years_passed = 100
		is_regular_empire = yes
		has_country_flag = BuyGrimoireFlag03
		has_country_flag = BuyAlchemyFlag03
		NOT = {
			has_country_flag = BuyNecronomiconFlag
		}
		NOT = {
			has_country_flag = DenyNecronomiconFlag
		}
	}
	## イベントの効果
	option = {
		allow = {
			owner = {
				resource_stockpile_compare = {
					resource = energy
					value >= 3500
				}
			}
		}
		name = magical_discoveries.8a.name
		set_country_flag = BuyNecronomiconFlag
		if = {
			limit = {
				NOT = {
					has_global_flag = LearnNecromancyFlag
				}
			}
			set_global_flag = LearnNecromancyFlag
		}
		owner = {
			add_resource = {
				energy = -3500
			}
		}
	}
	option = {
		name = magical_discoveries.8b.name
		set_country_flag = DenyNecronomiconFlag
	}
}

				}
			}

				id = magical_discoveries.102
			}
		}
	}

## 魔晄炉長期使用による星の不安定化6
planet_event = {
	id = magical_discoveries.107
	title = magical_discoveries.107.name
	desc = magical_discoveries.107.desc
	picture = GFX_evt_city_ruins
	show_sound = event_ghost_town
	hide_window = no
	is_triggered_only = yes
	trigger = {
		NOT = {
			has_planet_flag = building_magical_furnace_fase_4
		}
	}
	immediate = {
		destroy_colony = yes
		remove_planet_flag = building_magical_furnace_fase_0
		remove_planet_flag = building_magical_furnace_fase_1
		remove_planet_flag = building_magical_furnace_fase_2
		if = {
			limit = {
				NOR = {
					is_planet_class = pc_ringworld_habitable
					is_planet_class = pc_habitat
				}
			}
			random_list = {
				50 = {
					change_pc = pc_barren
				}
				50 = {
					change_pc = pc_barren_cold
				}
			}
			reset_planet = yes
			add_modifier = {
				modifier = "terraforming_candidate"
				days = -1
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_ringworld_habitable
			}
			change_pc = pc_ringworld_habitable_damaged
			reset_planet = yes
		}
		else_if = {
			limit = {
				is_planet_class = pc_habitat
			}
			remove_planet = yes
		}
	}
}
